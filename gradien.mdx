# Gradient Descent

Gradient Descent is an optimization algorithm used to find the values of parameters (coefficients) of a function (f) that minimizes a cost function (cost).


```python
import numpy as np
import matplotlib.pyplot as plt

def f(x):
    return x**2 + 5*np.sin(x)

def grad_f(x):
    return 2*x + 5*np.cos(x)

def gradient_descent(x0, learning_rate, n_iterations):
    x = [x0]
    for i in range(n_iterations):
        x_new = x[-1] - learning_rate*grad_f(x[-1])
        if abs(grad_f(x_new)) < 1e-3:
            break
        x.append(x_new)
    return x

x0 = -5
learning_rate = 0.1
n_iterations = 100

x = np.linspace(-5, 5, 100)
y = f(x)

x_min = gradient_descent(x0, learning_rate, n_iterations)[-1]
y_min = f(x_min)

plt.figure(figsize=(10, 5))
plt.plot(x, y, lw=2, label='f(x)')
plt.plot(x_min, y_min, 'ro', label='Minimum')
plt.xlabel('x')
plt.ylabel('y')
plt.legend()
plt.show()
```
